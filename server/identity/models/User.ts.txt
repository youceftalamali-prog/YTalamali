export enum AuthProvider {
  EMAIL = "email",
  GOOGLE = "google",
  GITHUB = "github",
  FACEBOOK = "facebook",
}

export enum UserRole {
  ADMIN = "admin",
  USER = "user",
  MANAGER = "manager",
}

export enum UserStatus {
  ACTIVE = "active",
  PENDING = "pending",
  SUSPENDED = "suspended",
  DELETED = "deleted",
}

export interface User {
  id: string;

  firstName: string;
  lastName: string;

  email: string;
  passwordHash: string;

  avatar?: string;

  authProvider: AuthProvider;
  providerId?: string;

  emailVerified: boolean;

  role: UserRole;
  status: UserStatus;

  createdAt: Date;
  updatedAt: Date;

  lastLoginAt?: Date;
}
